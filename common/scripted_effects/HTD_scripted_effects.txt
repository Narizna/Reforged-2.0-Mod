d_debug_HTD = {
	HTD_open_window = yes 
	
	for_each_loop = {
		array = HTD_armor
		
		meta_effect = {
			text = {
				HTD_[X]_[Y] = yes 
			}
			X = "[This.GetTag]"
			Y = "[?v.GetTokenKey]"
		}
	}
}

HTD_close_window = {

	clear_array = HTD_armor
	clr_country_flag = show_HTD
	
}

HTD_tech_effect = {
	hidden_effect = {
		if = {
			limit = {
				OR = {
					meta_trigger = {
						text = {
							NOT = { has_country_flag = HTD_[X]_[Y] }
							HTD_[X]_[Y] = yes
						}
						X = "[?HTD_target.GetTag]"
						Y = "[?v.GetTokenKey]"
					}
					has_country_flag = HTD_skip_trigger
				}
			}
			set_temp_variable = { xp_payment = xp_cost }
			multiply_temp_variable = { xp_payment = -1 }
			army_experience = xp_payment
			
			meta_effect = {
				text = {
					HTD_[X]_[Y] = yes 
					set_country_flag = HTD_[X]_[Y]
				}
				X = "[?HTD_target.GetTag]"
				Y = "[?v.GetTokenKey]"
			}
		}
	}
	effect_tooltip = {
		meta_effect = {
			text = {
				HTD_[X]_[Y] = yes 
			}
			X = "[?HTD_target.GetTag]"
			Y = "[?v.GetTokenKey]"
		}
	}
	
	add_to_variable = { ui_HTD = 1 }
}
